# Terminal Census

A capability matrix generated by running standardized probes against each terminal backend.
Run `bun census` to regenerate the data.

<script setup>
import { ref, computed } from 'vue'
import { data } from './census.data'

const categoryFilter = ref('all')
const filter = ref('')

const categoryNames = computed(() => {
  const cats = new Set()
  for (const id of data.features) {
    cats.add(id.split('.')[0])
  }
  return ['all', ...Array.from(cats).sort()]
})

const filteredFeatures = computed(() => {
  let ids = data.features
  if (categoryFilter.value !== 'all') {
    ids = ids.filter(id => id.startsWith(categoryFilter.value + '.'))
  }
  if (filter.value) {
    const q = filter.value.toLowerCase()
    ids = ids.filter(id => id.toLowerCase().includes(q))
  }
  return ids
})

/** Group filtered features by category for section headers. */
const groupedFeatures = computed(() => {
  const groups = []
  let currentCat = null
  for (const id of filteredFeatures.value) {
    const cat = id.split('.')[0]
    if (cat !== currentCat) {
      groups.push({ category: cat, features: [id] })
      currentCat = cat
    } else {
      groups[groups.length - 1].features.push(id)
    }
  }
  return groups
})

function getResult(backend, feature) {
  return data.results[backend]?.[feature]
}

function getNote(backend, feature) {
  return data.notes[backend]?.[feature] ?? ''
}

function backendSummary(name) {
  const results = data.results[name] ?? {}
  const entries = Object.values(results)
  const total = entries.length
  const yes = entries.filter(v => v === true).length
  const pct = total > 0 ? Math.round((yes / total) * 100) : 0
  return { total, yes, no: total - yes, pct }
}

/** Pretty-print feature id: "sgr.underline.curly" -> "underline.curly" (strip category prefix) */
function featureLabel(id) {
  const parts = id.split('.')
  return parts.slice(1).join('.')
}

/** Pretty-print category name */
function categoryLabel(cat) {
  const labels = {
    cursor: 'Cursor',
    erase: 'Erase',
    extensions: 'Extensions',
    modes: 'Modes',
    reset: 'Reset',
    scrollback: 'Scrollback',
    sgr: 'SGR (Styling)',
    text: 'Text',
  }
  return labels[cat] ?? cat
}

/** Pretty-print backend name */
function backendLabel(name) {
  const labels = {
    'ghostty-native': 'Ghostty',
    'xtermjs': 'xterm.js',
    'vt100': 'vt100',
    'vt100-rust': 'vt100-rust',
    'wezterm': 'WezTerm',
    'alacritty': 'Alacritty',
    'kitty': 'Kitty',
  }
  return labels[name] ?? name
}
</script>

## Backend Summary

<div class="census-summary">

    <span class="census-backend-name">{{ backendLabel(b.name) }}</span>
    <span class="census-version">{{ b.version }}</span>
    <span class="census-bar">
      <span class="census-bar-fill" :style="{ width: backendSummary(b.name).pct + '%' }"></span>
    </span>
    <span class="census-stats">
      {{ backendSummary(b.name).yes }}/{{ backendSummary(b.name).total }}
      ({{ backendSummary(b.name).pct }}%)
    </span>
  </div>
</div>

## Feature Matrix

<div class="census-filters">
  <label>
    Category:
    <select v-model="categoryFilter">
      <option v-for="cat in categoryNames" :key="cat" :value="cat">
        {{ cat === 'all' ? 'All' : categoryLabel(cat) }}
      </option>
    </select>
  </label>
  <label>
    Search:
    <input v-model="filter" type="text" placeholder="Filter features..." />
  </label>
</div>

<div class="census-table-wrap">
  <table class="census-table">
    <thead>
      <tr>
        <th class="census-feature-col">Feature</th>
        <th v-for="b in data.backends" :key="b.name">{{ backendLabel(b.name) }}</th>
      </tr>
    </thead>
    <tbody v-for="group in groupedFeatures" :key="group.category">
      <tr class="census-category-row">
        <td :colspan="data.backends.length + 1" class="census-category-header">
          {{ categoryLabel(group.category) }}
        </td>
      </tr>
      <tr v-for="id in group.features" :key="id">
        <td class="census-feature-name">{{ featureLabel(id) }}</td>
        <td
          v-for="b in data.backends"
          :key="b.name"
          :class="getResult(b.name, id) === true ? 'census-yes' : getResult(b.name, id) === false ? 'census-no' : 'census-unknown'"
          :title="getNote(b.name, id)"
        >
          <span v-if="getResult(b.name, id) === true" class="census-icon census-icon-yes">&#x2713;</span>
          <span v-else-if="getResult(b.name, id) === false" class="census-icon census-icon-no">&#x2717;</span>
          <span v-else class="census-icon census-icon-unknown">?</span>
        </td>
      </tr>
    </tbody>
  </table>
</div>

<p class="census-footer">
