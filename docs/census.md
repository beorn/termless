# Terminal Census

A capability matrix generated by running standardized probes against each backend.

Run `bun census` to regenerate the data.

<script setup>
import { ref, onMounted, computed } from 'vue'

const data = ref(null)
const error = ref(false)
const filter = ref('')
const categoryFilter = ref('all')

onMounted(async () => {
  try {
    const res = await fetch('/census/results/current.json')
    if (!res.ok) throw new Error('Not found')
    data.value = await res.json()
  } catch {
    error.value = true
  }
})

const categories = computed(() => {
  if (!data.value) return []
  const cats = new Set()
  for (const backend of data.value.backends) {
    const results = data.value.results[backend] ?? {}
    for (const id of Object.keys(results)) {
      const cat = id.split('-')[0]
      if (cat) cats.add(cat)
    }
  }
  return ['all', ...Array.from(cats).sort()]
})

const features = computed(() => {
  if (!data.value) return []
  const seen = new Set()
  for (const backend of data.value.backends) {
    const results = data.value.results[backend] ?? {}
    for (const id of Object.keys(results)) {
      seen.add(id)
    }
  }
  let ids = Array.from(seen).sort()
  if (categoryFilter.value !== 'all') {
    ids = ids.filter(id => id.startsWith(categoryFilter.value + '-'))
  }
  if (filter.value) {
    const q = filter.value.toLowerCase()
    ids = ids.filter(id => {
      const desc = data.value.descriptions[id] || ''
      return id.toLowerCase().includes(q) || desc.toLowerCase().includes(q)
    })
  }
  return ids
})

function getResult(backend, feature) {
  return data.value?.results[backend]?.[feature] ?? { support: 'error' }
}

function supportIcon(support) {
  if (support === 'yes') return '\u2705'
  if (support === 'partial') return '\u26A0\uFE0F'
  if (support === 'no') return '\u274C'
  return '\u2753'
}

function supportClass(support) {
  if (support === 'yes') return 'census-yes'
  if (support === 'partial') return 'census-partial'
  if (support === 'no') return 'census-no'
  return 'census-error'
}

function backendSummary(backend) {
  const results = data.value?.results[backend] ?? {}
  const entries = Object.values(results)
  const total = entries.length
  const yes = entries.filter(r => r.support === 'yes').length
  const partial = entries.filter(r => r.support === 'partial').length
  const no = entries.filter(r => r.support === 'no').length
  const pct = total > 0 ? Math.round((yes / total) * 100) : 0
  return { total, yes, partial, no, pct }
}
</script>

  </label>
